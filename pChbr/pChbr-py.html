クラス定義

class Rectangle:
  def __init__(self): # コンストラクタ
    self.height = 0
    self.width = 0

  def set(self, a, b):
    self.height = a
    self.width = b

使い方

rect = Rectangle()
rect.set(1,2)
print(rect.height) # 1

ウィンドウを作成

import tkinter.font as font

frm = tkinter.Tk()
geo = '600x400' # ウィンドウのサイズ

frm.geometry(geo)
frm.title(config.window.name)
fontC = font.Font(family='Helvetica', size=20, weight='bold')
tkinter.Label(
  frm,
  text='Text', # テキスト
  fg='#64ec86', # 文字色
  bg='#ffff00', # 文字の背景色
  font=fontC
).pack(side='top')
button = tkinter.Button(frm)  # ボタン作成
button["text"] = 'Click here!' # ボタンのテキスト設定
button.pack(side='top') # ボタンの設置位置変更
        
frm.mainloop()
